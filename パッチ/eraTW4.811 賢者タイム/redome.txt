アナタの現在の精力が上限値の一割以下なら、読書・学習の効率が上昇するようになるパッチです。

●COMF412 勉強.ERB
＞79行目から追記
;賢者タイムは学習がはかどる
SIF (BASE:MASTER:精力 *10) <= MAXBASE:MASTER:精力
	EXP:MASTER:学習経験 ++
	
●COMF436 読書.ERB
＞6行目から変数設定
#DIM READING_EXP
READING_EXP = 1
;賢者タイムは読書がはかどる
SIF (BASE:MASTER:精力 *10) <= MAXBASE:MASTER:精力
	READING_EXP = 2

＞86行目から改変
SELECTCASE TFLAG:193
	CASE 1
		EXP:MASTER:戦闘経験 += 5
	CASE 3
	;文学
		EXP:MASTER:学習経験 += READING_EXP
	CASE 5
	;実用
		SELECTCASE RAND:7
			CASE 0
				EXP:MASTER:会話経験 += READING_EXP
			CASE 1
				EXP:MASTER:撮影経験 += READING_EXP
			CASE 2
				EXP:MASTER:料理経験 += READING_EXP
			CASE 3
				EXP:MASTER:釣り経験 += READING_EXP
			CASE 4
				EXP:MASTER:採集経験 += READING_EXP
			CASE 5
				EXP:MASTER:調合経験 += READING_EXP
			CASEELSE
				EXP:MASTER:伐採経験 += READING_EXP
		ENDSELECT
	CASE 6
		CALL RECOVER(MASTER,100,"精力")
		CALL RECOVER(MASTER,100,"勃起")
ENDSELECT
